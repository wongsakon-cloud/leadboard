วิธีแก้ไขปัญหา Google Apps Script "Failed to Fetch"

สาเหตุหลักมักมาจากระดับสิทธิ์เข้าถึง (Permission Access) และการทำงานของ CORS เครือข่าย

1. แก้ไขหน้าต่างตั้งค่าสิทธิ์ตอนDeploy (สำคัญที่สุด ⭐)

หากพบข้อความว่า Failed to fetch แสดงว่าตอนที่คุณ Deploy ลิงก์ API คุณอาจไม่ได้เปิดสิทธิ์ให้ใครก็ได้เข้าถึงภายนอก วิธีแก้ไข:

  • เข้าไปที่ Google Apps Script ของคุณ
  • กดปุ่ม Deploy (การทำให้ใช้งานได้) > Manage deployments
  • คลิกไอคอนดินสอเพื่อแก้ไข และดูตรงหัวข้อ "Who has access" (ผู้มีสิทธิ์เข้าถึง)
  • ปรับเปลี่ยนค่าให้เป็น "Anyone" (ทุกคน) เสมอ (หากเลือกเป็น Only myself เว็บภายนอกจะไม่สามารถดึงข้อมูลได้)
  • กด Deploy อีกครั้ง แล้วคัดลอกลิงก์ Web App URL ใหม่มาใช้แทน

2. ตัวอย่างโครงสร้างโค้ดฝั่ง Google Apps Script (GAS) ที่หลีกเลี่ยงข้อจำกัด CORS

โครงสร้างฟังก์ชัน doGet() ของคุณ ควรส่งคืนค่าในรูปแบบ ContentService.MimeType.JSON โดยให้คัดลอกตัวอย่างนี้ไปประยุกต์ใช้เพื่อความแม่นยำสูง: 

function doGet(e) {
  try {
    var dateParam = e.parameter.date; // ดึงพารามิเตอร์วันที่ เช่น 2026-06-20
    var data = getDailyScheduleData(dateParam); // ฟังก์ชันดึงตารางเวรตามวันที่ของคุณ
    
    // ส่งข้อมูลออกเป็น JSON Text Output เพื่อหลีกเลี่ยงปัญหา CORS
    return ContentService.createTextOutput(JSON.stringify(data))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ error: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
